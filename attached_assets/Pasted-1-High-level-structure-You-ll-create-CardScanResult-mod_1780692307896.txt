1. High‑level structure
You’ll create:

CardScanResult – model for the JSON.

CardScanService – uploads a captured image to your backend.

CameraView – UIKit camera wrapper using AVCaptureSession.

ScanView – SwiftUI screen that shows the camera and calls the service.

Assumptions:

Backend URL (for now): http://localhost:8000/identify-card.

When you move to Railway later, you’ll just change this string.

3. CardScanService – upload JPEG to Flask
swift
import UIKit

final class CardScanService {
    // Change this later to your Railway URL
    private let endpoint = URL(string: "http://localhost:8000/identify-card")!

    func identifyCard(from image: UIImage) async throws -> CardScanResult {
        guard let jpegData = image.jpegData(compressionQuality: 0.8) else {
            throw URLError(.cannotEncodeContentData)
        }

        var request = URLRequest(url: endpoint)
        request.httpMethod = "POST"

        // Multipart/form-data boundary
        let boundary = "Boundary-\(UUID().uuidString)"
        request.setValue("multipart/form-data; boundary=\(boundary)", forHTTPHeaderField: "Content-Type")

        var body = Data()

        // File field "image" must match Flask's request.files["image"]
        body.append("--\(boundary)\r\n".data(using: .utf8)!)
        body.append("Content-Disposition: form-data; name=\"image\"; filename=\"scan.jpg\"\r\n".data(using: .utf8)!)
        body.append("Content-Type: image/jpeg\r\n\r\n".data(using: .utf8)!)
        body.append(jpegData)
        body.append("\r\n".data(using: .utf8)!)

        body.append("--\(boundary)--\r\n".data(using: .utf8)!)

        request.httpBody = body

        let (data, response) = try await URLSession.shared.data(for: request)

        guard let http = response as? HTTPURLResponse,
              (200..<300).contains(http.statusCode) else {
            throw URLError(.badServerResponse)
        }

        return try JSONDecoder().decode(CardScanResult.self, from: data)
    }
}
4. CameraView – UIKit camera wrapper
This is a lean UIViewControllerRepresentable that:

Shows live camera preview.

Captures a still when you tap the “Capture” button in SwiftUI.

swift
import SwiftUI
import AVFoundation

struct CameraView: UIViewControllerRepresentable {
    final class Coordinator: NSObject, AVCapturePhotoCaptureDelegate {
        var parent: CameraView

        init(parent: CameraView) {
            self.parent = parent
        }

        func photoOutput(_ output: AVCapturePhotoOutput,
                         didFinishProcessingPhoto photo: AVCapturePhoto,
                         error: Error?) {
            guard error == nil,
                  let data = photo.fileDataRepresentation(),
                  let image = UIImage(data: data) else {
                return
            }
            parent.onPhotoCaptured(image)
        }
    }

    let onPhotoCaptured: (UIImage) -> Void

    func makeCoordinator() -> Coordinator {
        Coordinator(parent: self)
    }

    func makeUIViewController(context: Context) -> UIViewController {
        let controller = CameraViewController()
        controller.captureDelegate = context.coordinator
        return controller
    }

    func updateUIViewController(_ uiViewController: UIViewController, context: Context) {}
}

final class CameraViewController: UIViewController {
    var captureSession: AVCaptureSession?
    var photoOutput = AVCapturePhotoOutput()
    var previewLayer: AVCaptureVideoPreviewLayer?
    var captureDelegate: AVCapturePhotoCaptureDelegate?

    override func viewDidLoad() {
        super.viewDidLoad()
        setupSession()
    }

    private func setupSession() {
        let session = AVCaptureSession()
        session.beginConfiguration()

        guard let device = AVCaptureDevice.default(.builtInWideAngleCamera,
                                                   for: .video,
                                                   position: .back),
              let input = try? AVCaptureDeviceInput(device: device),
              session.canAddInput(input) else {
            return
        }
        session.addInput(input)

        guard session.canAddOutput(photoOutput) else { return }
        session.addOutput(photoOutput)

        session.commitConfiguration()

        let preview = AVCaptureVideoPreviewLayer(session: session)
        preview.videoGravity = .resizeAspectFill
        preview.frame = view.bounds
        view.layer.addSublayer(preview)

        self.captureSession = session
        self.previewLayer = preview

        session.startRunning()
    }

    func capturePhoto() {
        let settings = AVCapturePhotoSettings()
        photoOutput.capturePhoto(with: settings, delegate: captureDelegate!)
    }

    override func viewDidLayoutSubviews() {
        super.viewDidLayoutSubviews()
        previewLayer?.frame = view.bounds
    }
}
To trigger capture from SwiftUI, we’ll use an @Environment hack: we’ll expose a capture() method through the view’s UIViewControllerRepresentable by adding an ObservableObject wrapper. To keep this answer focused, we can instead use a simple pattern: store a weak reference to the controller.

Modify CameraView like this:

swift
struct CameraView: UIViewControllerRepresentable {
    final class Coordinator: NSObject, AVCapturePhotoCaptureDelegate {
        var parent: CameraView

        init(parent: CameraView) {
            self.parent = parent
        }

        func photoOutput(_ output: AVCapturePhotoOutput,
                         didFinishProcessingPhoto photo: AVCapturePhoto,
                         error: Error?) {
            guard error == nil,
                  let data = photo.fileDataRepresentation(),
                  let image = UIImage(data: data) else {
                return
            }
            parent.onPhotoCaptured(image)
        }
    }

    let onPhotoCaptured: (UIImage) -> Void
    @Binding var controllerRef: CameraViewController?

    func makeCoordinator() -> Coordinator {
        Coordinator(parent: self)
    }

    func makeUIViewController(context: Context) -> CameraViewController {
        let controller = CameraViewController()
        controller.captureDelegate = context.coordinator
        DispatchQueue.main.async {
            controllerRef = controller
        }
        return controller
    }

    func updateUIViewController(_ uiViewController: CameraViewController, context: Context) {}
}
5. ScanView – SwiftUI screen that ties it together
swift
import SwiftUI

struct ScanView: View {
    @State private var cameraController: CameraViewController?
    @State private var isUploading = false
    @State private var result: CardScanResult?
    @State private var errorMessage: String?

    private let service = CardScanService()

    var body: some View {
        VStack(spacing: 16) {
            ZStack {
                CameraView(onPhotoCaptured: handleCapturedImage,
                           controllerRef: $cameraController)
                    .ignoresSafeArea(edges: .top)

                if isUploading {
                    Color.black.opacity(0.4)
                    ProgressView("Identifying card…")
                        .padding()
                        .background(Color.white)
                        .cornerRadius(8)
                }
            }
            .frame(height: 400)

            Button {
                cameraController?.capturePhoto()
            } label: {
                Text("Capture Card")
                    .font(.headline)
                    .foregroundColor(.white)
                    .frame(maxWidth: .infinity)
                    .padding()
                    .background(Color.blue)
                    .cornerRadius(12)
                    .padding(.horizontal)
            }

            if let result {
                VStack(alignment: .leading, spacing: 4) {
                    Text(result.name)
                        .font(.title2).bold()
                    Text("\(result.game.capitalized) • \(result.set)")
                        .foregroundColor(.secondary)
                    Text(String(format: "Confidence: %.2f", result.confidence))
                        .font(.footnote)
                        .foregroundColor(.secondary)
                }
                .padding(.horizontal)
            }

            if let errorMessage {
                Text(errorMessage)
                    .foregroundColor(.red)
                    .font(.footnote)
                    .padding(.horizontal)
            }

            Spacer()
        }
    }

    private func handleCapturedImage(_ image: UIImage) {
        Task {
            await identify(image: image)
        }
    }

    @MainActor
    private func identify(image: UIImage) async {
        isUploading = true
        errorMessage = nil
        defer { isUploading = false }

        do {
            let res = try await service.identifyCard(from: image)
            result = res
        } catch {
            errorMessage = "Failed to identify card: \(error.localizedDescription)"
        }
    }
}
